We implement a basic scatterplot with the code below. Lines marked with ðŸš§ can be improved.

<script>
data = [{ x: 10, y: 8},
        { x: 12, y: 6},
        { x: 24, y: 12}];

var margin = { top: 50, left: 50, bottom: 50, right: 50 };
var width = 400 - margin.left - margin.right;
var height = 400 - margin.top - margin.bottom;

var svg = d3.select('#chart')
  .attr('width', width + margin.left + margin.right)
  .attr('height', height + margin.top + margin.bottom)
  .append('g').attr('transform', 'translate(' + margin.left + ', ' + margin.top + ')');

var y = d3.scaleLinear()
  .domain([6, 12])  //ðŸš§ hard-coded values
  .range([height, 0]);

var x = d3.scaleLinear()
  .domain([10, 24])  //ðŸš§ hard-coded values
  .range([0, width]);

//plot circles after the axes so they appear on top
svg.selectAll('circle')
  .data(data)
  .enter()
  .append('circle')
  .attr('r', 3)
  .attr('cx', d => x(d.x))
  .attr('cy', d => y(d.y))
  .attr('class', 'dot');

var xAxis = d3.axisBottom()
  .scale(x); //ðŸš§ tick labels need formatting

svg.append('g')
  .attr('transform', 'translate(0,' + height + ')')
  .call(xAxis);

var yAxis = d3.axisLeft()
  .scale(y); //ðŸš§ tick labels need formatting

svg.append('g')
  .attr('class', 'axis')
  .call(yAxis);
</script>

2. Scatterplot improvements

We improve the scatterplot by:

  • Removing hard coded values in the axis domain with array.map() and d3.min, d3.max, d3.extent (see d3-array)
  • Adjusting the domains so that there are no data points on top of the axes
  • Formatting the tick labels (see continuous.ticks, continuous.tickFormat and d3.format)

The improved scatterplot code is shown below. 
<svg id="chart2"></svg>

<script>
data = [{ x: 10, y: 8},
        { x: 12, y: 6},
        { x: 24, y: 12}];

var margin = { top: 50, left: 50, bottom: 50, right: 50 };
var width = 400 - margin.left - margin.right;
var height = 400 - margin.top - margin.bottom;

var svg = d3.select('#chart2')
  .attr('width', width + margin.left + margin.right)
  .attr('height', height + margin.top + margin.bottom)
  .append('g').attr('transform', 'translate(' + margin.left + ', ' + margin.top + ')');

var PAD = 1;
//var yDomain = d3.extent(data.map( d => d.y ));
var yDomain = d3.extent(data, d => d.y);
var y = d3.scaleLinear()
  .domain([yDomain[0] - PAD, yDomain[1] + PAD])  //ðŸš§ hard-coded values
  .range([height, 0]);

var xDomain = d3.extent(data.map( d => d.x ));
var x = d3.scaleLinear()
  .domain([xDomain[0] - PAD, xDomain[1] + PAD])  //ðŸš§ hard-coded values
  .range([0, width]);

//plot circles after the axes so they appear on top
svg.selectAll('circle')
  .data(data)
  .enter()
  .append('circle')
  .attr('r', 3)
  .attr('cx', d => x(d.x))
  .attr('cy', d => y(d.y))
  .attr('class', 'dot');

var xAxis = d3.axisBottom()
  .scale(x) //ðŸš§ format tick labels
  .ticks(3)
  .tickFormat(d3.format('.2f'));

svg.append('g')
  .attr('transform', 'translate(0,' + height + ')')
  .call(xAxis);

var yAxis = d3.axisLeft()
  .scale(y) //ðŸš§ format tick labels
  .ticks(5)
  .tickFormat(d3.format('.2s')) 

svg.append('g')
  .attr('class', 'axis')
  .call(yAxis);
</script>
